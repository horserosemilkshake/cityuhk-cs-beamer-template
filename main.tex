\begin{frame}{Formula Template: Core Model}
\textbf{General objective}
\begin{equation}
\mathcal{L}(\theta) = \frac{1}{N}\sum_{i=1}^{N} \ell\bigl(f_{\theta}(x_i), y_i\bigr) + \lambda\,\Omega(\theta)
\end{equation}

\textbf{Symbol definitions}
\begin{itemize}
  \item $x_i$: input sample, $y_i$: target label.
  \item $f_{\theta}(\cdot)$: model with parameters $\theta$.
  \item $\ell(\cdot)$: task loss, $\Omega(\theta)$: regularization term.
\end{itemize}
\end{frame}

\begin{frame}{Multiline Equation Example: MLE Derivation}
\textbf{Assume i.i.d. Gaussian samples:} $x_i \sim \mathcal{N}(\mu,\sigma^2)$
\begin{align}
L(\mu) &= \prod_{i=1}^{n} \frac{1}{\sqrt{2\pi\sigma^2}}\exp\!\left(-\frac{(x_i-\mu)^2}{2\sigma^2}\right), \\
\ell(\mu) &= \log L(\mu)
= -\frac{n}{2}\log(2\pi\sigma^2) - \frac{1}{2\sigma^2}\sum_{i=1}^{n}(x_i-\mu)^2, \\
\frac{\partial \ell}{\partial \mu}
&= \frac{1}{\sigma^2}\sum_{i=1}^{n}(x_i-\mu)=0
\Rightarrow \hat{\mu}_{\text{MLE}} = \frac{1}{n}\sum_{i=1}^{n}x_i.
\end{align}
\end{frame}
