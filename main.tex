\begin{frame}{Formula Template: Core Model}
\textbf{General objective}
\begin{equation}
\mathcal{L}(\theta) = \frac{1}{N}\sum_{i=1}^{N} \ell\bigl(f_{\theta}(x_i), y_i\bigr) + \lambda\,\Omega(\theta)
\end{equation}

\textbf{Symbol definitions}
\begin{itemize}
  \item $x_i$: input sample, $y_i$: target label.
  \item $f_{\theta}(\cdot)$: model with parameters $\theta$.
  \item $\ell(\cdot)$: task loss, $\Omega(\theta)$: regularization term.
\end{itemize}
\end{frame}

\begin{frame}{Equation Template: Update Rules}
\textbf{State transition and parameter update}
\begin{align}
h_t &= \phi\left(W_h h_{t-1} + W_x x_t + b\right), \\
\hat{y}_t &= g\left(W_o h_t\right), \\
\theta_{t+1} &= \theta_t - \eta\,\nabla_{\theta} \mathcal{L}(\theta_t)
\end{align}
